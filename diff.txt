SELECT ?relatedEntity ?relatedEntityLabel ?property ?propertyLabel
WHERE {
    wd:Q22686 ?wdProperty ?relatedEntity.
    ?relatedEntity rdfs:label ?relatedEntityLabel .
    FILTER (lang(?relatedEntityLabel) = "en")

    ?property wikibase:directClaim ?wdProperty .
    {
        SELECT ?wdProperty (COUNT(?related) AS ?count)
        WHERE {
            wd:Q22686 ?wdProperty ?related .
        }
        GROUP BY ?wdProperty
        HAVING (COUNT(?related) = 1)
    }
    SERVICE wikibase:label { bd:serviceParam wikibase:language "en". }
}
LIMIT 100
